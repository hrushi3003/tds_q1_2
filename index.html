Code Sample: Compute NPV (Python)

from typing import Iterable

def npv(rate: float, cashflows: Iterable[float]) -> float:
    """Net Present Value given a discount rate and cash flow series."""
    return sum(cf / ((1 + rate) ** t) for t, cf in enumerate(cashflows, start=1))

# Example usage
print(npv(0.10, [100, 120, 150, 200]))
